1) Write a program to accepts two numbers from stdin and find all the odd as well as even
numbers present in between them.

import java.util.*;
import java.lang.*;
import java.io.*;


class acad
{
	public static void main (String[] args) throws java.lang.Exception
	{
		// your code goes here
		Scanner sc=new Scanner(System.in);
		int a=sc.nextInt();
		int b=sc.nextInt();
		int j=0,k=0;
		int[] even=new int[10];
		int[] odd=new int[10]; 
		for(int i=a+1;i<b;i++)
		{
			if(i%2==0)
			{
				even[j]=i;
				j++;
			}	
			else
			{
				odd[k]=i;
				k++;}
		}
		System.out.println("Even numbers are:");
		for(int i=0;i<j;i++)
		System.out.println(even[i]);
		System.out.println("Odd numbers are:");
		for(int i=0;i<k;i++)
		System.out.println(odd[i]);
	}
}
output:
1
5
Even numbers are:
2
4
Odd numbers are:
3

3) Write a program consisting method sum() and demonstrate the concept of method
overloading using this method.



import java.util.*;
import java.lang.*;
import java.io.*;


class acad
{
	public static void main (String[] args) throws java.lang.Exception
	{
		// your code goes here
		Scanner sc=new Scanner(System.in);
		
		System.out.println("Sum of two numbers:"+sum(1,19));  
		System.out.println("Sum of three numbers:"+sum(21,31,41));  
		
	}
	static int sum(int a,int b){
		return a+b;
		}  
static int sum(int a,int b,int c){
	return a+b+c;
	}  
}
output:
Sum of two numbers:20
Sum of three numbers:93
